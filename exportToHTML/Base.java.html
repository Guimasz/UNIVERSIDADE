public class application.Base implements application.IBase{
    private List<entities.Laboratorio> laboratorios;
    private List<entities.Departamento> departamentos;
    private List<entities.Professor> professores;
    private List<entities.Disciplina> disciplinas;
    private List<entities.Aluno> alunos;

    public application.Base() {
        laboratorios = new ArrayList<>();
        departamentos = new ArrayList<>();
        professores = new ArrayList<>();
        disciplinas = new ArrayList<>();
        alunos = new ArrayList<>();
        inicializarDados();
    }

    public List<entities.Laboratorio> getLaboratorios() {
        return laboratorios;
    }
    public List<entities.Departamento> getDepartamentos() {
        return departamentos;
    }
    public List<entities.Professor> getProfessores() {
        return professores;
    }
    public List<entities.Disciplina> getDisciplinas() {
        return disciplinas;
    }
    public List<entities.Aluno> getAlunos() {
        return alunos;
    }
    private void inicializarDados() {
        laboratorios.add(new entities.Laboratorio(1, "LAMI1", 30, true));
        laboratorios.add(new entities.Laboratorio(2, "LAMI2", 20, true));
    }



}
